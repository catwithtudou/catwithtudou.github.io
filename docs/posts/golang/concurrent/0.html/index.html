type LKQueue struct{
	head unsafe.Pointer
	tail unsafe.Pointer
}

type node struct{
	value interface{}
	next unsafe.Pointer
}


func NewLKQueue()*LKQueue{
	n:=unsafe.Pointer(&node{})
	return &LKQueue{
		head: n,
		tail: n,
	}
}

func (q *LKQueue)Enqueue(v interface{}){
	n:=&node{value: v}
	for{
		tail:=load(&q.tail)

		head := load(&q.head)
		tail := load(&q.tail)

		if head == load(&q.head){

func load(p *unsafe.Pointer)(n *node){
